def initialize
  @options = {
    'Image' => '',
    'name' => '',
    'Hostname' => '0.0.0.0',
    'ExposedPorts' => {},
    'Env' => [],
    'HostConfig' => {
      'PortBindings' => {},
      'Links' => []
    }
  }
end

def expose(port)
  @options['ExposedPorts']["#{port}/tcp"] = {}
  @options['HostConfig']['PortBindings']["#{port}/tcp"] = [{'HostPort' => port}]
end
